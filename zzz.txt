SELECT
    matatuCompanies.id,
    matatuCompanies.name,
    matatuCompanies.latitude,
    matatuCompanies.longitude,
    matatuCompanies.city,
    managers.first_name,
    managers.last_name,
    managers.username
FROM matatuCompanies
INNER JOIN managers
ON matatuCompanies.id = managers.company_id;
